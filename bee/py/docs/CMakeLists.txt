INCLUDE(FindDoxygen)
IF (DOXYGEN)
    ADD_CUSTOM_TARGET( pydocs ALL ${DOXYGEN} "${CMAKE_CURRENT_SOURCE_DIR}/Doxyfile" WORKING_DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR})
ELSE (DOXYGEN)
    MESSAGE(STATUS "WARNING: Doxygen not found")
ENDIF (DOXYGEN)
